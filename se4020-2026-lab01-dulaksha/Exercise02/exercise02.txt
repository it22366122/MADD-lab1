import SwiftUI

struct ContentView: View {
    @State var labelText: String = "Initial Value"
    var body: some View {
        VStack {
            Text(labelText)
                .font(.title)
            Button(action : {
                self.labelText = "Hello, World!"
            }){
                Text("Click Me")
            }
        }
        .padding()
    }
}
